\section{Keyboard shortcuts}

\subsection{General}
\begin{tabular}{@{}r@{\;}l@{}}
	\texttt{Tab}: &			completion in the local namespace \\
	\texttt{Ctrl-m s}: &	save notebook \\
	\texttt{Ctrl-m h}: &	show keyboard shortcuts \\
	\texttt{Ctrl-m l}: & 	toggle line numbers \\
	\texttt{Ctrl-r}: & 		search history for lines that match what is typed \\
\end{tabular}

\subsection{Editing}
\begin{tabular}{@{}r@{\;}l@{}}
	\texttt{Ctrl-m x}: &	cut cell \\
    \texttt{Ctrl-m c}: &	copy cell \\
	\texttt{Ctrl-m v}: &	paste cell \\
	\texttt{Ctrl-m d}: &	delete cell \\
	\texttt{Ctrl-m z}: &	undo last cell deletion \\
\end{tabular}

\subsection{Cell management}
\begin{tabular}{@{}r@{\;}l@{}}
	\texttt{Ctrl-m -}: &	split cell \\
	\texttt{Ctrl-m a}: &	insert cell above $\uparrow$ \\
	\texttt{Ctrl-m b}: &	insert cell below $\downarrow$ \\
	\texttt{Ctrl-m k}: &	move cell up $\uparrow$ \\
	\texttt{Ctrl-m j}: &	move cell down $\downarrow$ \\
	\texttt{Ctrl-m p}: &	select previous $\uparrow$ \\
	\texttt{Ctrl-m n}: &	select next $\downarrow$ \\
\end{tabular}

\subsection{Cell formatting}
\begin{tabular}{@{}r@{\;}l@{}}
	\texttt{Ctrl-m y}: &	code cell \\
	\texttt{Ctrl-m m}: &	markdown cell \\
	\texttt{Ctrl-m t}: &	raw cell \\
	\texttt{Ctrl-m 1-6}: &	heading 1--6 cell \\
\end{tabular}

\subsection{Execution control}
\begin{tabular}{@{}r@{\;}l@{}}
	\texttt{Shift-Enter}: &	run cell \\
	\texttt{Ctrl-Enter}: &	run cell in-place \\
	\texttt{Alt-Enter}: &	run cell, insert below \\	
	\texttt{Ctrl-m i}: &	interrupt kernel \\
	\texttt{Ctrl-m .}: &	restart kernel \\
	\texttt{Ctrl-m o}: &	toggle output \\
	\texttt{Ctrl-m 0}: &	toggle output scroll \\
\end{tabular}
